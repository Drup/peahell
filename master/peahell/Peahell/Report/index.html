Module Peahell.Report

Error and information reporting

val printf : ?loc:Location.loc -> ?sub:Location.loc list -> ('a, Stdlib.Format.formatter, unit, unit) Stdlib.format4 -> 'a
val fprintf : ?loc:Location.loc -> ?sub:Location.loc list -> Stdlib.Format.formatter -> ('a, Stdlib.Format.formatter, unit, unit) Stdlib.format4 -> 'a
val infof : ?loc:Location.loc -> ?sub:Location.loc list -> string -> ('a, Stdlib.Format.formatter, unit, unit) Stdlib.format4 -> 'a
val warnf : ?loc:Location.loc -> ?sub:Location.loc list -> string -> ('a, Stdlib.Format.formatter, unit, unit) Stdlib.format4 -> 'a
type msg = (Stdlib.Format.formatter -> unit) Location.t
type report_kind =
| Error
| Warning of string
| Info of string
| Output
type report = {
kind : report_kind;
main : msg;
sub : Location.loc list;
}
val errorf : ?loc:Location.loc -> ?sub:Location.loc list -> ('a, Stdlib.Format.formatter, unit, report) Stdlib.format4 -> 'a
val register_report_of_exn : (exn -> report option) -> unit
val report_exception : Stdlib.Format.formatter -> exn -> unit
val fail : ?loc:Location.loc -> ('a, Stdlib.Format.formatter, unit, 'b) Stdlib.format4 -> 'a

fail ~loc "msg" raises a built-in generic failure, for convenience.
